/*Circle Mouse Toy:http://actionsnippet.com/?paged=2*/var circles = [];for (var i = 0; i<30; i++){	var c = {};	c.x = stage.stageWidth / 2;	c.y = stage.stageHeight / 2;	c.scaleX = 1 + i/2;	c.scaleY = 0.5 + i/4;	circles.push(c);}function onEnterFrame() {     Pen.clear();     ScreenBuffer.fillRect(ScreenBuffer.rect,0xffffff);	circles[0].y += (stage.mouseY - circles[0].y) / 4;	for (var i = 1; i<circles.length; i++){		var pre = circles[i - 1];		circles[i].y += (pre.y - circles[i].y) / 4;           drawCircle(circles[i]);	}    ScreenBuffer.draw(Canvas);}function drawCircle(circle){		Pen.lineStyle(0,0x000000);		Pen.drawCircle(256,circle.y,10*circle.scaleX);}
